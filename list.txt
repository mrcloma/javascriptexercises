			matJogoVelha[0] = [0,0,0]
			matJogoVelha[1] = [0,0,0]
			matJogoVelha[2] = [0,0,0]
			
			/*Variavel que controla o jogador atual
			* false = jogador 1
			* true = jogador 2

			var jogador = false

			var jogoVencido = false

			var jogoMensagem = "" 
			
			/*Este metodo inicia o jogo*/
			function reinicio()
			{
				for(i =0; i < 3 ; i++)
				{
					for(j = 0; j < 3; j++)
					{
						matJogoVelha[i][j] = 0;
					}
				}
				jogador = false
				jogoVencido = false
				jogoMensagem = ""
				atualizaTela()
			}
			
			/*Este metodo atualiza a tela a partir da matriz
			* para utilizar este metodo deve-se conhecer o nome dos ID de cada celula da tabela html
			*/
			function atualizaTela()
			{
				var ids=1
				
				for(i =0; i < 3 ; i++)
				{
					for(j = 0; j < 3; j++)
					{
						//este metodo do document obtem o objeto html atraves do seu id, permitindo alterar atributos do mesmo
						//neste caso, altera a imagem de cada celula conforme o jogador
						celula = document.getElementById(ids)

						{
							celula.innerHTML =  "&nbsp" 
						}
						else if (matJogoVelha[i][j] == 1)
						{
						    celula.innerHTML = "<img src='bola.png'>"
						}
						else if (matJogoVelha[i][j] == 2)
						{
						    celula.innerHTML = "<img src='xix.png'>"
						}
						
						ids++
						
					}
				}
				//atualiza a informacao do jogador atual na tela

			function validaNumero(evt)
			{  
				/* 
					ASCII Código   - 48 a 57 = Teclas 0 a 9 
					ASCII Código   - 13 = Tecla Enter   
					ASCII Código   - 08 = Tecla Backspace  
					keyCode Código - 37 = Seta Esquerda 
					keyCode Código - 39 = Seta Direita 
					keyCode Código - 46 = Tecla Delete 
				*/  
            var tecla;  
      
            if(evt.which)   //firefox 
            {  
               tecla = evt.which  
            }  
            else if(evt.keyCode) 
            {  
               tecla = evt.keyCode;  
            }  

            if(tecla==13 || tecla == 8 || tecla == 37 || tecla == 39 || tecla == 46 || (tecla >= 48 && tecla <=57)){  
			  return true;  
            }  
            else  
            {  
              return false;  
            }  
          }  
		  
		  
		  function validaNumeroComTamanho(evt, obj)
			{  
				/* 
					ASCII Código   - 48 a 57 = Teclas 0 a 9 
					ASCII Código   - 13 = Tecla Enter   
					ASCII Código   - 08 = Tecla Backspace  
					keyCode Código - 37 = Seta Esquerda 
					keyCode Código - 39 = Seta Direita 
					keyCode Código - 46 = Tecla Delete 
				*/  
            var tecla;  
			
			//var obj = document.getElementById("campoData");
      
            if(evt.which)   //firefox 
            {  
               tecla = evt.which  
            }  
            else if(evt.keyCode) 
            {  
               tecla = evt.keyCode;  
            }  
			
            if(tecla==13 || tecla == 8 || tecla == 37 || tecla == 39 || tecla == 46 || (tecla >= 48 && tecla <=57)){  
			  if (obj.value.length < 10)
				return true;  
			  else
				return false;
            }  
            else  
            {  
              return false;  
            }  
          } 
		  
		  function aceitaData(obj)
		  {
			//Este evento é executado quando o controle perde o foco
			//valida a data atraves do objeto date
			//o metodo new Date() quando recebe um texto, converte o mesmo para uma data
			//se o texto nao for uma data, ele sempre retorna a mensagem "Invalid Date", que pode ser comparada 
			//para validacao
			if (obj.value != "" )
			{
				var d = new Date(obj.value);
				
				if (d.toString() == "Invalid Date")
				{
					alert("Data Inválida!");
				}
			}
		  }
		  
		  function validaData(obj)
		  {
				//verifica o tamanho do campo
				if(obj.value.length == 2 ||
				   obj.value.length == 5)
				{
					obj.value = obj.value + "/";
				}
		  }
